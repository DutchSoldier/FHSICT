CREATE OR REPLACE PROCEDURE "GETALLECATEGORIEEN" (
  "CATEGORIECURSOR" OUT SYS_REFCURSOR) IS
BEGIN 
   OPEN CATEGORIECURSOR FOR SELECT * FROM Categorie;
END;
/
CREATE OR REPLACE PROCEDURE "VERDUBBEL" (
  "TEVERDUBBELEN" IN NUMBER, 
  "RESULTAAT" OUT NUMBER) IS
BEGIN 
	dbms_output.put_line('Getal wordt verdubbeld.');
	RESULTAAT := TEVERDUBBELEN * 2;
END;
/
